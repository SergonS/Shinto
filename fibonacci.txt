program fibonacci;

int function fibonacci(n: int)
{
    var int a, b, c, i;

    if (n == 1)
    {
        output(0);
    }

    a = 0;
    b = 1;

    output(a);
    output(b);

    i = 3;

    while (i <= n)
    {
        c = a + b;
        output(c);
        a = b;
        b = c;
        i = i + 1;
    }
    return 0;
}

function main()
{
    var int num;

    output("Enter the number of items to generate the Fibonacci series");
    input(num);
    fibonacci(num);
}
